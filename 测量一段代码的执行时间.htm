rdtsc指令返回的是自开机始CPU的周期数，返回的是一个64位的值EDX：EAX（高32在EDX，低32位在EAX）。可以利用这条指令，测试一段代码的执行效率。

 

 

#define LODWORD(_qw)    ((DWORD)(_qw))

 

#define HIDWORD(_qw)    ((DWORD)(((_qw) >> 32) & 0xffffffff))

 

typedef unsigned __int64 QWORD;

 

QWORD st0,en0;

 

__asm

 

{

 

cpuid

 

rdtsc

 

mov dword ptr[st0],eax

 

mov dword ptr[st0+4],edx

 

}

 

Sleep(500);

 

__asm

 

{

 

cpuid

 

rdtsc

 

mov dword ptr[en0],eax

 

mov dword ptr[en0+4],edx

 

}

 

en0=en0-st0;

 

int va0=LODWORD(en0);

 

int va1=HIDWORD(en0);

 

printf("%x%x\n",va1,va0);

 

 

测量一段代码的执行时间

 

LARGE_INTEGER litmp;

 

LONGLONG QPart1, QPart2;

 

double dfMinus, dfFreq, dfTim;

 

SetThreadAffinityMask(GetCurrentThread(),0x00000001);

 

QueryPerformanceFrequency(&litmp);         //获取CPU时钟频率

 

dfFreq = (double)litmp.QuadPart;

 

QueryPerformanceCounter(&litmp);         //获取开始计数值

 

QPart1 = litmp.QuadPart;

 

Sleep(100);

 

QueryPerformanceCounter(&litmp);        //获取结束计数值

 

QPart2 = litmp.QuadPart;

 

dfMinus = (double)(QPart2 - QPart1);

 

dfTim = dfMinus/dfFreq; //计算持续时间，单位为秒误差不超过1微妙

 

printf("%f\n",dfTim);
